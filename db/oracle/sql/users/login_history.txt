Show SESSION login history
select c.username,a.SAMPLE_TIME, a.SQL_OPNAME, a.SQL_EXEC_START, a.program, a.module, a.machine, b.SQL_TEXT
from DBA_HIST_ACTIVE_SESS_HISTORY a, dba_hist_sqltext b, dba_users c
where a.SQL_ID = b.SQL_ID(+)
and a.user_id=c.user_id
and c.username='&username'
order by a.SQL_EXEC_START asc;

from dba_audit_session

---AUDIT SESSION PELO HORARIO DO ERRO
select *
from   sys.dba_audit_session das
where  das.timestamp > to_date('09/16/2022 17:00:00', 'MM/DD/YYYY HH24:MI:SS')
and    das.timestamp < to_date( '09/16/2022 17:30:00','MM/DD/YYYY HH24:MI:SS') +1
order  by timestamp

----AUDIT SESSION AUDIT TRAIL
select * from dba_audit_trail dat where returncode=1017 
and  dat.timestamp > to_date('09/16/2022 16:00:00', 'MM/DD/YYYY HH24:MI:SS')
and    dat.timestamp < to_date( '11/09/2022 01:00:00','MM/DD/YYYY HH24:MI:SS') +1
order by timestamp ASC;

----VERIFICA COMANDO DE ALTER USER / CREATE USER / DROP USER
SELECT *
FROM sys.aud$
WHERE action# IN (
SELECT action 
FROM audit_actions
WHERE name IN 
('ALTER USER',
'CREATE USER',
'DROP USER'));
