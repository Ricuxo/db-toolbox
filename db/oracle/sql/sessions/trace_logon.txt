CREATE OR REPLACE TRIGGER "owner"."TRACE_LOGON"
AFTER LOGON ON owner.SCHEMA
DECLARE
  vTeste varchar2(20);
BEGIN
  EXECUTE IMMEDIATE 'alter session set sql_trace true';
  EXECUTE IMMEDIATE 'alter session set timed_statistics=true';
  EXECUTE IMMEDIATE 'alter session set max_dump_file_size=16384';
  EXECUTE IMMEDIATE 'alter session set tracefile_identifier=TRACE_owner';
  EXECUTE IMMEDIATE 'alter session set events ''10046 trace name context forever, level 12''';
  select 'Usuario OWNER' into vTeste from dual;
EXCEPTION
   WHEN OTHERS
   THEN
      raise_application_error (-20001, 'Problem: ' || SQLERRM);
END;
